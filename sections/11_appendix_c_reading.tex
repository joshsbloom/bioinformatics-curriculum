\chapter{Appendix C: Reading List (Prioritized)}

\section{Start Here (the essentials)}
\begin{itemize}[nosep]
  \item StatQuest YouTube (Josh Starmer) --- watch the PCA, DESeq2, and p-value episodes first
  \item Software Carpentry: Unix Shell workshop (free, interactive, excellent)
  \item Scanpy PBMC3k tutorial (the canonical single-cell starting point)
  \item Marees et al.\ -- A tutorial on conducting GWAS (step-by-step)
  \item R for Data Science (r4ds.hadley.nz) --- free, skip to Ch.\ 3 if you know basics
\end{itemize}

\section{Go Deeper (when you're ready)}
\begin{itemize}[nosep]
  \item Buffalo -- \textit{Bioinformatics Data Skills} (the Unix + workflows bible)
  \item McKinney -- \textit{Python for Data Analysis}, 3rd Ed (the Pandas bible)
  \item Holmes \& Huber -- \textit{Modern Statistics for Modern Biology} (free online)
  \item Love, Huber, Anders -- DESeq2 paper (Genome Biology, 2014)
  \item Heumos et al.\ -- Best practices for single-cell analysis (Nat Rev Genet, 2023)
  \item Uffelmann et al.\ -- Genome-wide association studies (Nat Rev Methods Primers, 2021)
  \item Bulik-Sullivan et al.\ -- LD Score regression (Nature Genetics, 2015)
\end{itemize}

\section{Reference (keep on your shelf)}
\begin{itemize}[nosep]
  \item Gillespie -- \textit{Population Genetics: A Concise Guide}
  \item Efron -- \textit{Large-Scale Inference}
  \item Storey \& Tibshirani -- FDR paper (PNAS, 2003)
  \item Wang et al.\ -- SuSiE (JRSS-B, 2020)
  \item Wilson et al.\ -- Good enough practices in scientific computing (PLOS Comp Bio, 2017)
\end{itemize}

\section{Online Tools \& Databases}
\begin{itemize}[nosep]
  \item explainshell.com --- paste any Unix command to see what each flag does
  \item Bioconductor workflows --- bioconductor.org/packages/release/workflows/
  \item GWAS Catalog --- ebi.ac.uk/gwas/
  \item PGS Catalog --- pgscatalog.org
  \item nf-core --- nf-co.re (community Nextflow pipelines)
  \item Anthropic Prompt Engineering Guide --- docs.anthropic.com
\end{itemize}
